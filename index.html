async function connectWallet() { if (!window.keplr) { alert("Please install Keplr wallet"); return; } await window.keplr.enable("columbus-5"); // Terra Classic chain id const offlineSigner = window.keplr.getOfflineSigner("columbus-5"); const accounts = await offlineSigner.getAccounts(); const address = accounts[0].address; document.getElementById("walletAddress").innerText = address; }
